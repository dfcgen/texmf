%% Special commands/text/math in headings/titles
%% Usage: \headingspecial{cmds}{replacement}
%% Example: \headingspecial{\footnote{Test}}{x times y}
\providecommand{\headingspecial} [2] {#1}
\renewcommand{\headingspecial} [2] {\texorpdfstring{#1}{#2}}


%% Inline math in headings/titles
%% Usage: \headingmath{expr}{replacement}
%% Example: \headingmath{x^y}{x times y}
\providecommand{\headingmath} [2] {\ensuremath{#1}}
\renewcommand{\headingmath} [2] {%
  \headingspecial{%
    \mathversion{bold}\ensuremath{#1}\mathversion{normal}}%
    {#2}%
}
