\newcommand{\lps} {s} %% Laplace variable s or p

\newcommand{\acf} {\varphi} %% Auto (Cross) -correlation function
\newcommand{\spd} {S} %% Spectral Power Density
\newcommand{\drf} {D} %% Characteristic function (Drosselung)
\newcommand{\trf} {H} %% Transfer function
\newcommand{\atf} {A} %% Attenuation
\newcommand{\phf} {B} %% Phase (neg. angle)
\newcommand{\irf} {h} %% Impulse response function

\newcommand{\edrf} [3] [0] {\epq{#1}{\drf}{#2}{#3}} %%  elliptic \drf 


\DeclareMathOperator{\usf}   {s} %% Unit step function
\DeclareMathOperator{\rect}  {r} %% Rectangular pulse
\DeclareMathOperator{\dirac} {\delta} %% Dirac pulse
